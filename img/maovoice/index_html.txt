const VALID_ROOM = '840137';

function joinRoom() {
  const el = document.getElementById('room-input');
  const v = el.value.trim();
  if(v !== VALID_ROOM) {
    el.classList.add('shake');
    el.style.borderColor = 'var(--red)';
    el.value = '';
    el.placeholder = '房间不存在或已过期';
    setTimeout(() => {
      el.classList.remove('shake');
      el.style.borderColor = '';
      el.placeholder = '输入房间号…';
    }, 2000);
    return;
  }
  roomNum = v;
  document.getElementById('join-num').textContent = '#' + v;
  showScreen('joining');
  const user = USERS[0]; // always Alex K. for room 840137
  document.getElementById('ping-val').textContent = '—';
  let pingMs = Math.floor(Math.random()*18)+8;
  setTimeout(() => {
    document.getElementById('ping-val').textContent = pingMs + ' ms';
    document.getElementById('ping-val').style.color = 'var(--green)';
  }, 800);
  setTimeout(() => {
    document.getElementById('room-num-display').textContent = v;
    document.getElementById('call-room-num').textContent = '#' + v;
    // Avatar — фото или градиент+буква
    const setAvatar = (el, initialEl, user) => {
      if(user.avatar) {
        el.style.background = 'none';
        el.style.backgroundImage = `url(${user.avatar})`;
        el.style.backgroundSize = 'cover';
        el.style.backgroundPosition = 'center';
        initialEl.textContent = '';
      } else {
        el.style.background = user.bg;
        el.style.backgroundImage = '';
        initialEl.textContent = user.init;
      }
    };
    setAvatar(document.getElementById('room-avatar-el'), document.getElementById('room-initial'), user);
    document.getElementById('room-name-el').textContent = user.name;
    setAvatar(document.getElementById('remote-avatar'), document.getElementById('remote-initial-call'), user);
    document.getElementById('remote-name-call').textContent = user.name;
    showScreen('room');
    updateRoomScreenState(); // применяем офлайн-статус если уже был в звонке
  }, 2200);
}

// ── JOIN SOUND (Web Audio, без внешних файлов) ──
function playJoinSound() {
  try {
    const ctx = new (window.AudioContext || window.webkitAudioContext)();
    const master = ctx.createGain();
    master.gain.value = 0.45;
    master.connect(ctx.destination);

    // Две ноты: до → ми (восходящий чайм)
    const notes = [523.25, 659.25]; // C5, E5
    notes.forEach((freq, i) => {
      const osc = ctx.createOscillator();
      const env = ctx.createGain();
      osc.type = 'sine';
      osc.frequency.value = freq;
      env.gain.setValueAtTime(0, ctx.currentTime + i * 0.13);
      env.gain.linearRampToValueAtTime(1, ctx.currentTime + i * 0.13 + 0.01);
      env.gain.exponentialRampToValueAtTime(0.001, ctx.currentTime + i * 0.13 + 0.35);
      osc.connect(env);
      env.connect(master);
      osc.start(ctx.currentTime + i * 0.13);
      osc.stop(ctx.currentTime + i * 0.13 + 0.36);
    });

    // Лёгкий реверб — третья нота тише и чуть позже
    const osc3 = ctx.createOscillator();
    const env3 = ctx.createGain();
    osc3.type = 'sine';
    osc3.frequency.value = 783.99; // G5
    env3.gain.setValueAtTime(0, ctx.currentTime + 0.26);
    env3.gain.linearRampToValueAtTime(0.6, ctx.currentTime + 0.27);
    env3.gain.exponentialRampToValueAtTime(0.001, ctx.currentTime + 0.7);
    osc3.connect(env3);
    env3.connect(master);
    osc3.start(ctx.currentTime + 0.26);
    osc3.stop(ctx.currentTime + 0.71);

    setTimeout(() => ctx.close(), 1000);
  } catch(e) { console.warn('Join sound unavailable') }
}

let currentAudio = null;
let playlistTimeouts = [];
let currentVolume = 1.0;
let audioActive = false;

function startAudio() {
  audioActive = true;
  playlistTimeouts = [];

  const primer = new Audio();
  primer.volume = 0;
  primer.src = AUDIO_PLAYLIST[0].url;
  primer.load();
  const primePromise = primer.play();
  if(primePromise) primePromise.then(() => primer.pause()).catch(() => {});
  currentAudio = primer;
  primer.volume = currentVolume;

  scheduleTrack(0);
}

function scheduleTrack(index) {
  if(!audioActive) return;
  if(index >= AUDIO_PLAYLIST.length) return;
  const track = AUDIO_PLAYLIST[index];
  const t = setTimeout(() => {
    if(!audioActive) return;
    playTrack(track.url, index, () => scheduleTrack(index + 1));
  }, track.delayBefore);
  playlistTimeouts.push(t);
}

function playTrack(url, index, onEnd) {
  if(currentAudio) {
    currentAudio.pause();
    currentAudio.currentTime = 0;
    currentAudio.src = '';
    try { currentAudio.load(); } catch(e) {}
  }
  const audio = new Audio(url);
  audio.volume = currentVolume;
  currentAudio = audio;
  let done = false; // гарантируем что колбэк сработает ровно один раз
  const finish = () => {
    if(done) return;
    done = true;
    if(!audioActive) return;
    if(index === 1) triggerAlexMessage();
    onEnd();
  };
  console.log('[MaoVoice] playing track', index, url);
  audio.play().catch(e => console.warn('[MaoVoice] play failed:', e));
  audio.addEventListener('ended', () => { console.log('[MaoVoice] track ended', index); finish(); }, { once: true });
  audio.addEventListener('error', (e) => { console.warn('[MaoVoice] track error', index, e); finish(); }, { once: true });
}

function stopAudio() {
  audioActive = false;
  playlistTimeouts.forEach(clearTimeout);
  playlistTimeouts = [];
  if(currentAudio) {
    currentAudio.pause();
    currentAudio.currentTime = 0;
    currentAudio.src = '';
    try { currentAudio.load(); } catch(e) {}
    currentAudio = null;
  }
}

// ── WAVEFORM ──
function startWaveform() {
  const canvas=document.getElementById('wave-canvas');
  if(!canvas) return;
  const ctx=canvas.getContext('2d');
  const dpr=devicePixelRatio||1;
  canvas.width=canvas.offsetWidth*dpr;
  canvas.height=canvas.offsetHeight*dpr;
  ctx.scale(dpr,dpr);
  let ph=0;
  function draw(){
    const W=canvas.offsetWidth, H=canvas.offsetHeight;
    ctx.clearRect(0,0,W,H);
    const bars=22, gap=3;
    const bw=(W-(bars-1)*gap)/bars;
    const amp=.35+.65*Math.abs(Math.sin(ph*.28));
    for(let i=0;i<bars;i++){
      const t=i/bars;
      const h=amp*(0.25+0.75*Math.pow(Math.abs(Math.sin(t*Math.PI*4+ph)),1.2))*H*.9;
      const x=i*(bw+gap);
      const y=(H-h)/2;
      const alpha=0.3+amp*0.7;
      ctx.fillStyle=`rgba(35,165,90,${alpha})`;
      ctx.beginPath();
      ctx.roundRect(x,y,bw,h,2);
      ctx.fill();
    }
    ph+=.1;
    waveAF=requestAnimationFrame(draw);
  }
  draw();
}

function stopWaveform() { if(waveAF) cancelAnimationFrame(waveAF); }

function playTypingSound(durationMs) {
  try {
    const ctx = new (window.AudioContext || window.webkitAudioContext)();
    const clicks = Math.floor(durationMs / 80);
    for(let i = 0; i < clicks; i++) {
      const t = ctx.currentTime + i * 0.08 + Math.random() * 0.02;
      const buf = ctx.createBuffer(1, Math.floor(ctx.sampleRate * 0.03), ctx.sampleRate);
      const d = buf.getChannelData(0);
      for(let j = 0; j < d.length; j++) d[j] = (Math.random()*2-1) * Math.exp(-j/(ctx.sampleRate*0.008));
      const src = ctx.createBufferSource(); src.buffer = buf;
      const g = ctx.createGain(); g.gain.value = 0.07 + Math.random()*0.05;
      src.connect(g); g.connect(ctx.destination);
      src.start(t);
    }
    setTimeout(() => ctx.close(), durationMs + 500);
  } catch(e) {}
}
